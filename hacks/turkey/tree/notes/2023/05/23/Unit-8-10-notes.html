nums = ["10", "15", "20", "25", "30", "35"]
potentialMin = int(nums.pop())
while len(nums) != 0:
    newNum = int(nums.pop())
    if newNum < potentialMin:
        potentialMin = newNum
print(potentialMin, "is the minimum value.")
